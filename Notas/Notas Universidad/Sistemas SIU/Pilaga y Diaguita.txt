1) systemctl stop apache2.service   ->		Bajamos el apache.
2) systemctl stop postgresql		->		Bajamos la base.
3) Creamos un snapshot en VMWare.
3.1) systemctl start postgresql    ->		Levantamos el motor de la base
4) cd /root/Instalado/Pilaga/		->		Nos dirigmos a la carpeta donde se encuentra el paquete del cambio (SIU-Pilaga_3.10.3.zip  vendor.zip)
5) unzip vendor.zip					->		Descomprimimos ambos paquetes
6) unzip SIU-Pilaga_3.10.3.zip		
7) mv vendor/ SIU-Pilaga_3.10.3/		->		Movemos vendor/ a la carpeta SIU-Pilaga_3.10.3/
8) mv SIU-Pilaga_3.10.3 pilaga_3103 	->		Cambiamos de nombre la carpeta SIU-Pilaga
9) chown -R www-data:www-data pilaga_3103/ 		-> Damos permisos a usuario www-data
10) systemctl stop jasper.service		->		Stopeamos jasper (sistema de correos)
11) mv pilaga_3103/ /usr/local/siu/		->		Movemos la carpeta a /usr/local/siu/
12) su - www-data		->		Ingresamos con usuario www-data
13) cd /usr/local/siu/pilaga3103/		->		Nos dirigmos a la carpeta
14)	cp ../pilaga3102/instalador.env .		->		Copiamos el enviroment de la version anterior a /usr/local/siu/pilaga3103/
15) ./bin/instalador proyecto:actualizar --instalacion-anterior /usr/local/siu/pilaga3102		->		Corremos proyecto actualizar VERSION ANTERIOR 
15) ./bin/instalador instalacion:modo-mantenimiento --sin-mantenimiento		->		Sacamos el proyecto modo mantenimiento
16) dos2unix instalacion/toba.conf		->		Modificamos el formato del archivo
17) vi instalacion/toba.conf		->		Modicamos en el archivo las rutas con la nueva version e ingresamos las ip que permite el acceso
18) vi instalacion/i__produccion/instancia.ini		->		Dejamos url vacio y full_url = "https://test-pilaga.universidad.edu.ar" 
19) exit		->		Salimos de usuario www-data
20) ll /etc/init.d/jasper 		->		Verificamos que exista el link simbolico ( lrwxrwxrwx 1 root root 48 Jan 25 10:42 /etc/init.d/jasper -> /usr/local/siu/pilaga3103/bin/pilaga_reportes.sh* )
21) rm /etc/init.d/jasper		->		Eliminamos el link simbolico
22) ln -s /usr/local/siu/pilaga3103/bin/pilaga_reportes.sh /etc/init.d/jasper		->		Creamos el nuevo link simbolico con la nueva carpeta de la version
23) ll /etc/init.d/jasper 		->		Verificamos que se haya creado de manera correcta
24) vimdiff /usr/local/siu/pilaga3102/bin/pilaga_reportes.sh /usr/local/siu/pilaga3103/bin/pilaga_reportes.sh		->		Verificamos si hay modificaciones con la version anterior
25) netstat -npl		->		Verificamos que el puerto 8081 No se este utilizando ya que es el que utiliza jasper
26) chmod 755 /usr/local/siu/pilaga3120/bin/pilaga.sh	
28) chmod +x /usr/local/siu/pilaga3120/bin/pilaga_reportes.sh
29) chmod +x /usr/local/siu/pilaga3120/bin/toba_pilaga.sh
29.1) vimdiff /usr/local/siu/pilaga3111/bin/toba_pilaga.sh /usr/local/siu/pilaga3120/bin/toba_pilaga.sh   ->	Carpetas de version
30) systemctl start jasper.service		->		Levantamos jasper
31) netstat -npl		->		Verificamos que el puerto 8081 se este utilizando con java ya que se levanto jasper
32) vi /etc/apache2/sites-enabled/httpd-ssl.conf 		->		Modificamos el apache con la nueva carpeta pilaga3103/
33) cp /root/Instalado/Pilaga/siu_universidad.gif /usr/local/siu/pilaga3120/www/img/		->		Copiamos la imagen que se encuentra en /root/Instalado/Pilaga/siu_universidad.gif
34) systemctl start apache2.service 		->		iniciamos apache2.service
