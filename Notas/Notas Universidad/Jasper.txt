---------------------------------------------------------------
Jasper para inicializar una tarea:
---------------------------------------------------------------

Crear un archivo /etc/init.d/jasper con el siguiente contenido:

----------------------- ejemplo 1 -----------------------
#!/bin/bash
### BEGIN INIT INFO
# Provides:          diaguita_reportes
# Required-Start:    $syslog
# Required-Stop:     $syslog
# Default-Start:     2 3 4 5
# Default-Stop:      0 1 6
# Short-Description: Diaguita Reportes
# Description:       Servicio de reportes JasperReports para SIU-Diaguita
#                    Creado para SIU-Diaguita <diaguita@siu.edu.ar>
#
### END INIT INFO

NAME=diaguita-reportes
DESC="Servidor de Reportes para SIU-Diaguita"
CURRENT_DIR=$(dirname "$(realpath $0)")

# ruta a la instacion de SIU-Diaguita
# ATENCION: debe definir la ruta a la instalacion SIU-Diaguita
PATH_DIAGUITA="/usr/local/siu/diaguita330"

# ruta a JasperReports
# Las librerias se deben encontrar en toba/php/3ros
PATH_JASPER=$PATH_DIAGUITA"/vendor/siu-toba/jasper/JavaBridge/WEB-INF/lib/JavaBridge.jar"

# niveles aceptados: (3 = errores) (5 = depuracion)
DEBUG_LEVEL="3"

# archivo de log
DEBUG_FILE="/var/log/diaguita_reportes.log"

# con SERVLET_LOCAL, solo acepta conexiones "loopback" o de localhost
# con SERVLET, acepta conexiones de todas las interfaces de red
DAEMON="/usr/bin/java -- -Duser.language=es -Duser.country=AR -jar $PATH_JASPER SERVLET_LOCAL:8081 $DEBUG_LEVEL $DEBUG_FILE"

PIDFILE=/var/run/diaguita/$NAME.pid

if ! test -f $PATH_JASPER
then
        echo "Debe configurar la variable PATH_DIAGUITA correctamente"
        exit 1
fi

set -e

case "$1" in
      start)
          echo -n "Iniciando $DESC: " #--quiet
          start-stop-daemon --start --quiet --background --make-pidfile --pidfile $PIDFILE  --exec $DAEMON &
          echo "$NAME."
          ;;

      stop)
          echo -n "Deteniendo $DESC: "
          start-stop-daemon --oknodo --stop --pidfile $PIDFILE
          if [ -e $PIDFILE ]
            then rm $PIDFILE
          fi
          echo "$NAME."
          ;;
      restart|force-reload)
          $0 stop
          sleep 3
          $0 start
          ;;
      *)
        N=/etc/init.d/$NAME
        echo "Uso: $N {start|stop|restart|force-reload}" >&2
        exit 1
        ;;
esac

exit 0
