#!/bin/bash
TYPE=$1
NAME=$2
STATE=$3
SERVERNAME=$(hostname -s | tr '[:lower:]' '[:upper:]')
EMAIL_ALERT="usuario3.apellido@empresa.com"
EMAIL_SUMMARY="iti@empresa.com"
echo "1: $1   2: $2  3: $3" > /root/servicios.txt

case ${STATE} in
    "MASTER")
        [ -x /etc/keepalived/gateway ] && /etc/keepalived/gateway
        /root/serviciosarriba.sh
        echo "$SERVERNAME: $TYPE $NAME transition to $STATE STATE" | mail -E -s "$SERVERNAME: $TYPE $NAME transition to $STATE STATE" ${EMAIL_ALERT}
        if [ ! -f /tmp/keepalived_notify.alert ] || [ $(($(date +%Y%m%d%H%M)-$(date -r /tmp/keepalived_notify.alert +%Y%m%d%H%M))) -gt 1 ] ; then
            echo > /tmp/keepalived_notify.alert
            sleep 5 && for i in $(pgrep keepalived) ; do grep $i /var/log/messages | grep -e MASTER -e BACKUP ; done | mail -E -s "$SERVERNAME: Summary transition STATE" ${EMAIL_SUMMARY}
        fi
    ;;
    "BACKUP")
        /root/serviciosup.sh
    ;;
    "FAULT")
        echo "$SERVERNAME: $TYPE $NAME transition to $STATE STATE" | mail -E -s "$SERVERNAME: $TYPE $NAME transition to $STATE STATE" ${EMAIL_ALERT}
        /root/serviciosdown.sh
    ;;
    *)
        echo "$SERVERNAME: $TYPE $NAME transition to $STATE STATE" | mail -E -s "$SERVERNAME: $TYPE $NAME UNKNOWN STATE" ${EMAIL_ALERT}
        exit 1
    ;;
esac

exit 0

#!/bin/bash
 
export PATH="/usr/sbin:/usr/bin:/sbin:/bin:/usr/local/sbin:/usr/local/bin"
 
LOG="/root/servicios.log"
LOCK="/root/serviciosup.lock"
 
exec >> "$LOG" 2>&1
echo "[$(date +'%Y-%m-%d %H:%M')] Inicio de servicios UP"
 
# Lockfile
if [ -f "$LOCK" ]; then
    echo "Ya está corriendo otro proceso de serviciosup."
    ps axf | grep "$(cat "$LOCK")" | grep -v grep
    exit 1
else
    echo $$ > "$LOCK"
fi
 
# Recursos DRBD
declare -A recursos=(
    ["/home"]="particion"
#    ["/var/www"]="www1"
)
 
# Promover a primary y esperar dispositivos
for punto in "${!recursos[@]}"; do
    recurso="${recursos[$punto]}"
    dispositivo="/dev/drbd${recurso/particion/0}"
#    dispositivo="${dispositivo/www1/1}"
 
    echo "[$(date +'%Y-%m-%d %H:%M')] Promoviendo $recurso a Primary..."
 
# Esperar a que el peer esté en Secondary o no conectado
echo "Esperando que el peer de $recurso esté en Secondary o desconectado..."
for i in {1..30}; do
    estado=$(drbdadm status "$recurso" | grep -A1 "peer-disk" | tail -n1)
    echo "  → Estado peer: $estado"
    if echo "$estado" | grep -Eq 'Secondary|Diskless|Inconsistent|Consistent|Negotiating'; then
        break
    fi
    sleep 1
done
 
# Intentar promover a Primary
echo "Promoviendo $recurso a Primary..."
if ! drbdadm primary "$recurso" --force; then
    echo "ERROR: No se pudo promover $recurso a Primary. Verificá estado del nodo peer."
    drbdadm status "$recurso"
    rm -f "$LOCK"
    exit 1
fi
 
echo "[$(date +'%Y-%m-%d %H:%M')] $recurso en Primary"
    if ! drbdadm status "$recurso" | grep -q "role:Primary"; then
        echo "[$(date +'%Y-%m-%d %H:%M')] ERROR: No se pudo poner $recurso en Primary"
        rm -f "$LOCK"
        exit 1
    fi
 
    echo "[$(date +'%Y-%m-%d %H:%M')] Esperando a que $dispositivo esté disponible..."
    for j in {1..10}; do
        if [ -b "$dispositivo" ]; then
            echo "$dispositivo disponible"
            break
        fi
        sleep 1
    done
 
    if [ ! -b "$dispositivo" ]; then
        echo "ERROR: $dispositivo no apareció en /dev"
        drbdadm status "$recurso"
        rm -f "$LOCK"
        exit 1
    fi
 
    # Verificar sistema de archivos
    if ! blkid "$dispositivo" | grep -q "TYPE="; then
        echo "ERROR: $dispositivo no tiene sistema de archivos. ¿Se olvidaron de formatear?"
        rm -f "$LOCK"
        exit 1
    fi
 
    # Montar si no está montado
    if ! mountpoint -q "$punto"; then
        echo "Montando $dispositivo en $punto..."
        mount "$dispositivo" "$punto" && echo "$punto montado correctamente." || {
            echo "ERROR: Falló montaje de $punto"
            rm -f "$LOCK"
            exit 1
        }
    else
        echo "$punto ya está montado."
    fi
done
 
###### Iniciar servicios  ######
echo "Iniciando servicios..."

##Agregar servicios

# Nginx
systemctl start nginx.service
if ! systemctl is-active --quiet nginx.service; then
    echo "ERROR: nginx no se pudo iniciar"
    rm -f "$LOCK"
    exit 1
fi
 
# Limpieza final
echo "[$(date +'%Y-%m-%d %H:%M')] serviciosup completado correctamente. Borrando lockfile..."
rm -f "$LOCK"
exit 0

#!/bin/bash
 
export PATH="/usr/sbin:/usr/bin:/sbin:/bin:/usr/local/sbin:/usr/local/bin"
#PATH=/sbin:/bin:/usr/sbin:/usr/bin:/usr/local/sbin
#export PATH
 
sleep 2
LOG="/root/servicios.log"
LOCK="/root/serviciosdown.lock"
 
echo "serviciosdown" >> "$LOG"
echo "$(date +"%Y-%m-%d %H:%M") empezando a bajar todo..." | tee -a "$LOG"
 
# Lockfile
if [ -f "$LOCK" ]; then
    echo "ya está tratando de bajar servicios..." | tee -a "$LOG"
    ps axf | grep $(cat "$LOCK") | grep -v grep | tee -a "$LOG"
    exit 1
else
    echo "$(date +"%Y-%m-%d %H:%M") creando lockfile..." | tee -a "$LOG"
    echo $$ > "$LOCK"
fi
 
###### Parar servicios ######
echo "Parando servicios..." | tee -a "$LOG"

# Nginx
echo "Parando nginx..." | tee -a "$LOG"
systemctl stop nginx.service
if systemctl is-active --quiet nginx.service; then
    echo "ERROR: nginx sigue funcionando" | tee -a "$LOG"
    rm -f "$LOCK"
    exit 1
fi

##Agregar Servicios

 
# Recursos y puntos de montaje
declare -A recursos
recursos["/home"]="particion"
##recursos["/var/www"]="www1"
 
# Desmontar y bajar DRBD
for punto in "${!recursos[@]}"; do
    recurso="${recursos[$punto]}"
 
    echo "Procesando $punto -> DRBD $recurso" | tee -a "$LOG"
 
    if mountpoint -q "$punto"; then
        echo "Intentando desmontar $punto..." | tee -a "$LOG"
        if umount "$punto"; then
            echo "$(date +"%Y-%m-%d %H:%M") $punto desmontado correctamente." | tee -a "$LOG"
        else
            echo "No se pudo desmontar $punto" | tee -a "$LOG"
            df -h | grep "$punto" | tee -a "$LOG"
            rm -f "$LOCK"
            exit 1
        fi
    else
        echo "$punto ya está desmontado." | tee -a "$LOG"
    fi
 
    echo "Pasando $recurso a secundario..." | tee -a "$LOG"
    drbdadm secondary "$recurso"
    sleep 2
 
    if drbdadm status | grep "$recurso role:Secondary" | grep -v grep ; then
        echo "$(date +"%Y-%m-%d %H:%M") $recurso correctamente en Secondary." | tee -a "$LOG"
    else
        echo "Error: $recurso no pasó a Secondary correctamente." | tee -a "$LOG"
        drbdadm status | tee -a "$LOG"
        rm -f "$LOCK"
        exit 1
    fi
done
 
# Todo OK
echo "$(date +"%Y-%m-%d %H:%M") Todos los recursos desmontados y en Secondary. El otro nodo puede tomar el control." | tee -a "$LOG"
echo "borrando lockfile" | tee -a "$LOG"
rm -f "$LOCK"
exit 0

    device              /dev/drbd0;
    disk                /dev/sda1;
    meta-disk           internal;
    flexible-meta-disk  internal;

    options {
        auto-promote no;
    }

    net {
        protocol C;
        after-sb-0pri  discard-least-changes;
        after-sb-1pri  discard-secondary;
        after-sb-2pri  call-pri-lost-after-sb;
    }

    disk {
        on-io-error    detach;
        no-md-flushes;
        no-disk-flushes;
    }

    on n8n01a.linux.backend {
        address xx.xx.xx.xx;
    }

    on n8n01b.linux.backend {
