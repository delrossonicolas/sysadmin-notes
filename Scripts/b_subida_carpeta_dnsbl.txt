#!/bin/bash
## Sube dnsbl y preserva permisos de archivos existentes - Fecha 26 Mayo 2026
## Copia en el secundario, solo ejecutar en el primario - No hace backup
## Cambiar permisos desde carpeta de subida para ajustar (VERIFICAR PERMISOS ANTES DE SUBIR con text compare) Hacer en local lo siguiente:
##chmod 664 cache.txt check.spam.exim descargaSpamhaus.py destinatarios_rechazados helo_geo.exim limite.auth.exim rbl.conf.exim (hecho en local)
##chmod 755 cache_para_exim.sh confirme.ip.bash continente_ip.py helo.py hora-antes.bsh hora.bsh nombrearchivo.txt pais_ip.py separacion.bsh traer_remitentes_denegados.bsh verificar_rbl.py notifica_correo_ip.bsh  (hecho en local)


SERVIDORES=(

)

DESTINO_REMOTO="/etc/exim/dnsbl"

FALLIDOS=()

for SERVIDOR in "${SERVIDORES[@]}"; do
  echo -e "\033[1;33mProcesando servidor: $SERVIDOR\033[0m"

  # Sincronizar contenido (elimina del destino lo que no exista en origen)
  rsync -av --delete -e "ssh -o BatchMode=yes" \
    dnsbl/ "root@$SERVIDOR:${DESTINO_REMOTO}/" \
  && echo "Contenido sincronizado con éxito en $SERVIDOR" \
  || {
    echo "No se pudo sincronizar contenido en $SERVIDOR"
    FALLIDOS+=("$SERVIDOR")
    continue
 }
  # Ajustar permisos y ownership
  ssh -o BatchMode=yes "root@$SERVIDOR" "

    chown root:root ${DESTINO_REMOTO}
    chown root:root ${DESTINO_REMOTO}/*
    chown exim:exim ${DESTINO_REMOTO}/nombrearchivo.txt
    chown exim:exim ${DESTINO_REMOTO}/separacion.bsh
    chown exim:exim ${DESTINO_REMOTO}/confirme.ip.bash
    chmod 755 ${DESTINO_REMOTO}/confirme.ip.bash
    
  " || {
    echo "No se pudieron ajustar permisos en $SERVIDOR"
    FALLIDOS+=("$SERVIDOR")
    continue
  }

  # Reiniciar Exim
  ssh -o BatchMode=yes "root@$SERVIDOR" "
    if systemctl is-active --quiet exim.service; then
      echo 'Esperando 3 segundos antes de reiniciar Exim...'
      sleep 3
      systemctl restart exim.service && echo '[\$(hostname)] Exim reiniciado correctamente.'
    else
      echo '[\$(hostname)] Exim está detenido. No se reinicia.'
    fi
  " || {
    echo "No se pudo reiniciar Exim en $SERVIDOR"
    FALLIDOS+=("$SERVIDOR")
    continue
  }

  echo -e "\033[1;32mCompletado para $SERVIDOR\033[0m"
done

# Mostrar fallidos
if [ ${#FALLIDOS[@]} -gt 0 ]; then
  echo
  echo "Servidores con errores:"
  printf '%s\n' "${FALLIDOS[@]}"
fi
